hyperion::mpl#

hyperion::mpl is a C++20 meta programming library similar to Boost.Hana.

Itâ€™s currently under active development and has not yet hit a stable release point.

See the Quick Start Guide for how to get started using hyperion::mpl.

For an overview of each module, see the links in the left sidebar or below.

 1#include <hyperion/mpl/value.h>
 2#include <hyperion/mpl/type.h>
 3
 4#include <concepts>
 5
 6using namespace hyperion::mpl;
 7
 8constexpr auto val1 = Value<4>{};
 9constexpr auto val2 = Value<2>{};
10
11constexpr auto meaning_of_life = (val1 * 10_value) + val2;
12
13static_assert(meaning_of_life == 42);

15template<ValueType TValue>
16struct add_one {
17    static inline constexpr auto value = TValue::value + 1;
18};
19
20template<ValueType TValue>
21struct times_two {
22    static inline constexpr auto value = TValue::value * 2;
23};
24
25static_assert(2_value
26              .apply<add_one>()
27              .apply<times_two>()
28              .apply<add_one>() == 7);

30constexpr auto val3 = 10;
31static_assert(decltype_(val3)
32              .apply<std::remove_reference>()
33              .apply<std::remove_const>()
34              .apply<std::add_rvalue_reference>()
35              == decltype_<int&&>());

37constexpr auto add_const = [](MetaType auto type)
38    -> std::add_const<typename decltype(type)::type>
39{
40    return {};
41};
42
